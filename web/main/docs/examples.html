Code Examples

Learn Strata by reading and running these examples. You can copy them into the playground and modify them.

Arithmetic & Math

Basic Calculator

import io from std::io func add(a: int, b: int) => int { return a + b } func subtract(a: int, b: int) => int { return a - b } func multiply(a: int, b: int) => int { return a * b } io.print(add(10, 5)) io.print(subtract(10, 5)) io.print(multiply(10, 5))

Fibonacci Sequence

import io from std::io func fibonacci(n: int) => int { if (n <= 1) { return n } return fibonacci(n - 1) + fibonacci(n - 2) } for (var i: int = 0; i < 10; var i: int = i + 1) { io.print(fibonacci(i)) }

Factorial

import io from std::io func factorial(n: int) => int { if (n <= 1) { return 1 } return n * factorial(n - 1) } io.print(factorial(5)) // 120

Loops & Iteration

Sum of Numbers

import io from std::io func sum(n: int) => int { var total: int = 0 for (var i: int = 1; i <= n; var i: int = i + 1) { var total: int = total + i } return total } io.print(sum(100)) // 5050

Loop with Break

import io from std::io var i: int = 0 while (i < 100) { if (i == 10) { break } io.print(i) var i: int = i + 1 }

String Operations

String Length & Case

import io from std::io import text from std::text let message: string = "Strata Language" io.print(text.len(message)) io.print(text.upper(message)) io.print(text.lower(message))

Mathematical Functions

Using Math Module

import io from std::io import math from std::math io.print(math.sqrt(16.0)) // 4.0 io.print(math.pow(2.0, 3.0)) // 8.0 io.print(math.abs(-42)) // 42 io.print(math.floor(3.7)) // 3.0 io.print(math.ceil(3.2)) // 4.0

Arrays

Array Iteration

import io from std::io let numbers: [int] = [1, 2, 3, 4, 5] for (var i: int = 0; i < len(numbers); var i: int = i + 1) { io.print(numbers[i]) }

Type Safety

Type Checking Example

import io from std::io let x: int = 42 let y: float = 3.14 let z: string = "hello" io.print(x) io.print(y) io.print(z) // let wrong: int = z // ERROR: Type mismatch!
